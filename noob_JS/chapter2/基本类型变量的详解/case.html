基本类型的变量是存放再栈内存(Stack)里面的

var a,b;
a = "zyj";
b = a;
console.log(a); // zyj
console.log(b); // zyj
a = "呵呵"; // 改变 a 的值，并不影响 b 的值
console.log(a); // 呵呵
console.log(b); // zyj

引用类型的值是保存在堆内存（Heap）中的对象（Object）

var a = {name:"percy"};
var b;
b = a;
a.name = "zyj";
console.log(b.name); // zyj
b.age = 22;
console.log(a.age); // 22
var c = {
name: "zyj",
age: 22
};
